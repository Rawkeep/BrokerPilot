       ```jsx
       import clsx from 'clsx';

       export function GlassButton({ children, variant = 'default', className, disabled, ...props }) {
         return (
           <button
             className={clsx(
               'glass-button',
               variant === 'primary' && 'glass-button--primary',
               className
             )}
             disabled={disabled}
             {...props}
           >
             {children}
           </button>
         );
       }
       ```

    3. Create `client/src/components/ui/GlassInput.jsx`:
       ```jsx
       import clsx from 'clsx';

       export function GlassInput({ className, label, id, error, ...props }) {
         const inputId = id || props.name;
         return (
           <div className="glass-input-wrapper">
             {label && <label htmlFor={inputId} className="glass-input-label">{label}</label>}
             <input
               id={inputId}
               className={clsx('glass-input', error && 'glass-input--error', className)}
               {...props}
             />
             {error && <span className="glass-input-error">{error}</span>}
           </div>
         );
       }
       ```

    4. Create `client/src/hooks/useTheme.js` per RESEARCH.md System Theme Detection pattern (D-04):
       - Import useEffect from 'react'
       - Import useSettingsStore
       - useTheme() hook:
         - Reads theme, brokerType, setTheme from settingsStore
         - useEffect: resolves effective theme (if 'system', check window.matchMedia('(prefers-color-scheme: dark)'))
         - Sets document.documentElement.setAttribute('data-theme', effectiveTheme)
         - If brokerType is set, sets document.documentElement.setAttribute('data-broker', brokerType)
         - If theme === 'system', adds MediaQueryList 'change' event listener, cleans up on unmount
         - Returns { theme, setTheme, brokerType, effectiveTheme }
